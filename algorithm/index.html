栈 #

栈是什么？ #

  • 一个后进先出的数据结构。

  • js 中没有栈，但可以用Array实现栈的所有功能。

  • 栈的常用操作：push、pop、stack[stack.length-1]

  • 基于Array的栈结构

    js
    export class Stack {
    constructor() {
        this.items = []
    }
    // 向栈添加元素
    push(element) {
        this.items.push(element)
    }
    // 从栈移除元素
    pop() {
        return this.items.pop()
    }
    // 查看栈顶元素
    peek() {
        return this.items[this.items.length -1]
    }
    // 检查栈是否为空
    isEmpty() {
        // return Array.isEmpty(this.items)
        return this.items.length === 0
    }
    // 返回栈的长度
    size() {
        return this.items.length
    }
    // 清空栈元素
    clear() {
        this.items = []
    }
}

  • 基于Object实现的栈结构：

    使用基于数组的实现的方式占用太多的时间和空间，大部分方法的时间复杂度是O(n)；数组是一个有序集合，为了保证元素排列有序，会占用更多的内存空间。

    export class Stack {
    constructor() {
        this.count = 0 // 我们将使用一个count属性来帮助我们记录栈的大小（也能帮助我们从数据结构中添加和删除元素）

            return undefined

        delete this.items[this.count]
        return result

        if (this.isEmpty()) {
            return undefined
        }

        this.items = {}

        // }

            return ''

场景一：十进制转二进制 #

  • 后出来的余数反而要排到前面。

  • 把余数依次入栈，然后再出栈，就可以实现余数的倒序输出。

  • 代码：

    js
    /**
 * 十进制转二进制
 * @param {十进制数} decNumber 
 * @param {*使用的栈} Stack
 * @returns binaryString
 */
export function decimalToBinary(decNumber, Stack) {
    const remStack = new Stack()
    let number = decNumber
    let rem // 余数
    let binaryString = ''

    while (number > 0) { // 当除法的结果不为0时，可以得到余数
        rem = Math.floor(number % 2) // 取余数
        remStack.push(rem)
        number = Math.floor(number / 2) // 对商取整
    }

    while (!remStack.isEmpty()) {
        binaryString += remStack.pop().toString()
    }

    return binaryString
}
/**
 * 升级版，进制转换算法,支持任意进制转换
 * @param {需要转换的数字} decNumber 
 * @param {进制} base 
 * @param {栈结构} Stack 
 * @returns 
 */
export function baseConverter(decNumber, base, Stack) {
    const remStack = new Stack();
    const digits = '0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ';
    let decNumber = number;
    let rem;
    let baseString = '';
  
    if (!(base >= 2 && base <= 36)) {
      return '';
    }
  
    while (decNumber > 0) {
      rem = Math.floor(decNumber % base);
      remStack.push(rem);
      decNumber = Math.floor(decNumber / base);
    }
  
    while (!remStack.isEmpty()) {
      baseString += digits[remStack.pop()]; // 需要对栈中余数进行进制的转化
    }
  
    return baseString;
  }

有效的括号（20） #

  • 解题思路：

    1. 对于没有闭合的左括号而言，越靠后的左括号，对应的右括号越靠前。
    2. 满足后进先出，考虑用栈。

  • 解题步骤：

    1. 新建一个栈。
    2. 扫描字符串，遇到左括号入栈，遇到和栈顶括号类型匹配的右括号就出栈，类型不匹配直接判定为不合法。
    3. 最后栈空了就合法，否则不合法。

  • 代码：

    js
    /**
 * @param {string} s
 * @return {boolean}
 */
var isValid = function(s) {
    if(s.length % 2 === 1) return false // 优化，如果为奇数则不符合要求
    const stack = []
    const matchString = new Map([
        [')', '('],
        [']', '['],
        ['}', '{']
    ])
    for(let str of s) {
        const top = stack[stack.length - 1]
         if(matchString.has(str)) {
             if(matchString.get(str) === top) {
                 stack.pop()
             } else {
                 return false
             }
         } else {
             stack.push(str)
         }
    }
    return stack.length === 0
};

export default class Queue {
    constructor() {
        this.count = 0 // 声明一个count属性来帮助我们控制队列的大小
        this.lowestCount = 0 // 需要一个变量来帮助我们追踪第一个元素
        this.items = {}
    }

    // 向队列添加元素
    enqueue(element) {
        this.items[this.count] = element
        this.count++
    }

    // 从队列移除元素
    dequeue() {
        if (this.isEmpty()) {
            return undefined
        }
        const result = this.items[this.lowestCount]
        delete this.items[this.lowestCount]
        this.lowestCount++
        return result
    }

    // 查看队列头元素
    peek() {
        if (this.isEmpty()) {
            return undefined
        }
        return this.items[this.lowestCount]
    }

    // 检查队列是否为空并获取它的长度
    isEmpty() {
        return (this.count - this.lowestCount) === 0
        // 或者 this.size() === 0
    }
    size() {
        return this.count - this.lowestCount
    }

    // 清空队列
    // 要清空队列中的所有元素，我们可以调用dequeue方法直到它返回undefined，也可以简单地将队列中的属性值重设为和构造函数中的一样。
    clear() {
        this.items = {}
        this.count = 0
        this.lowestCount = 0
    }

    // 创建toString方法
    toString() {
        if (this.isEmpty()) {
            return ''
        }
        let objString = `${this.items[this.lowestCount]}`
        for (let i = this.lowestCount + 1; i < this.count; i++) {
            objString = `${objString}, ${this.items[i]}`
        }
        return objString
    }
}

循环队列 #

/**
 * 循环队列的应用
 * @param {入队的数据}} elementsList 
 * @param {循环次数} num 
 * @param {队列结构} Queue 
 * @returns 
 */
function hotPotato(elementsList, num, Queue) {
    const queue = new Queue()
    const elimminatedList = []
    elementsList.forEach((cv, index) => queue.enqueue(cv)); // 把数据全部入队
    while(queue.size() > 1) {
        for(let i = 0; i < num; i++) {
            queue.enqueue(queue.dequeue()) // 从队列开头移除一项，再添加到队列末尾。（循环队列）
            elimminatedList.push(queue.dequeue()) // 淘汰移除队列头部的元素
        }
    }
    return {
        elimminated: elimminatedList,
        winner: queue.dequeue() // 输出队列最后一个元素
    }
}

双端队列 #

双端队列是什么？ #

  • 双端队列是一种允许我们同时从前端和后端添加和移除元素的特殊队列。
  • 同时遵守了“先进先出”和“后进后出”原则
  • 是队列和栈相结合的一种数据结构。

基于对象实现的双端队列 #

js
/*
 * 双端队列是一种特殊的队列
 */
export default class Deque {
    constructor() {
        this.count = 0
        this.lowestCount = 0
        this.items = {}
    }
    
    addFront(element) {
        if (this.isEmpty()) {
          this.addBack(element);
        } else if (this.lowestCount > 0) { // 一个元素已经从前端移除了，此时lowestCount是大于等于1的
          this.lowestCount--;
          this.items[this.lowestCount] = element;
        } else { // lowestCount为0的情况（模拟数组）（也可以直接使负值）
          for (let i = this.count; i > 0; i--) {
            this.items[i] = this.items[i - 1];
          }
          this.count++;
          this.items[0] = element;
        }
    }

    addBack(element) {
        this.items[this.count] = element;
        this.count++;
    }

    removeFront() {
        if (this.isEmpty()) {
          return undefined;
        }
        const result = this.items[this.lowestCount];
        delete this.items[this.lowestCount];
        this.lowestCount++;
        return result;
    }

    removeBack() {
        if (this.isEmpty()) {
          return undefined;
        }
        this.count--;
        const result = this.items[this.count];
        delete this.items[this.count];
        return result;
    }

    peekFront() {
        if (this.isEmpty()) {
          return undefined;
        }
        return this.items[this.lowestCount];
    }
    
    peekBack() {
        if (this.isEmpty()) {
            return undefined;
        }
        return this.items[this.count - 1];
    }

    // 检查队列是否为空并获取它的长度
    isEmpty() {
        return this.size() === 0
    }
    size() {
        return this.count - this.lowestCount
    }

    // 清空队列
    clear() {
        this.items = {}
        this.count = 0
        this.lowestCount = 0
    }

    // 创建toString方法
    toString() {
        if (this.isEmpty()) {
            return ''
        }
        let objString = `${this.items[this.lowestCount]}`
        for (let i = this.lowestCount + 1; i < this.count; i++) {
            objString = `${objString}, ${this.items[i]}`
        }
        return objString
    }
}

回文检查器 #

  • 最简单的方式：将字符串方向排列并检查它和原字符串是否相同。
  • 最简单的数据结构解决方法：双端队列。
js
/**
 * 利用双向队列，实现回文检查器
 * @param {字符串} str 
 * @param {双向队列} Dque 
 * @returns 
 */
function palindromeChecker(str, Dque) {
    // 检验字符串是否合法
    if(str == null || (str !== null && str.length === 0)) {
        return false
    }
    const deque = new Dque()
    const lowerString = str.lowerString().split(' ').join('')
    let isEqual = true
    let firstChar, lastChar
    for (let i = 0; i < lowerString.length; i++) {
        deque.addBack(lowerString.charAt(i))
    }
    while(deque.size() > 1 && isEqual) {
        firstChar = deque.removeFront()
        lastChar = deque.removeFrontBack()
        if(firstChar !== lastChar) {
            isEqual = false
        }
    }
    return isEqual
}

链表 #

链表是什么？ #

  • 多个元素组成的列表。
  • 元素存储不连续，之间通过用next指针连在一起。
  • 链表常用操作：修改 next、遍历链表。

数组vs链表 #

  • 数组：增删非首尾元素时，往往需要移动元素。
  • 链表：增删非首尾元素，不要移动元素，只需要更改 next 的指向即可。

JS 中的链表

  • JS 中没有链表。

  • 可以用 Object 模拟链表。

  • 链表常用操作：修改 next、遍历链表。

删除链表中的节点（237） #

  • 思路：

    • 无法直接获取被删除节点的上个节点。
    • 将被删除的节点，转移到下个节点。

  • 解题步骤：

    2. 删除下个节点。

    • 空间：O(1)

    // 思路：将被删除节点的值改为下个节点的值；再删除下个节点
    node.val = node.next.val
    node.next = node.next.next
};

反转链表（206） #

  • 思路：

    • 反转两个节点：将 n+1 的next指向n。
    • 反转多个节点：双指针遍历链表，重复上述操作。

  • 解题步骤：

    1. 双指针一前一后遍历链表。
    2. 反转双指针。

  • 代码：

    • 迭代版本

    • 时间：O(n)

    • 空间：O(1)

      js
      /**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var reverseList = function(head) {
    // 双指针一前一后遍历链表
    // 反转双指针
    let p1 = head
    let p2 = null
    while(p1) {
        // 进行链表反转操作
        const _nxet = p1.next
        p1.next = p2
        // 循环继续走下去
        p2 = p1
        p1 = _nxet
    }
    return p2  // 返回反转后的链表(此时p2指向的是链表head p1指向null)
};

    • 递归版本：

      js
      /**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var reverseList = function(head) {
    if (head == null || head.next == null) {//递归终止条件
        return head;
    }
    const newHead = reverseList(head.next);//递归调用reverseList
    //如果链表是 1->2->3->4->5，那么此时的newHead就是5
    //而head是4，head的下一个是5，下下一个是空
    //所以head.next.next 就是5->4
    head.next.next = head; // (连转反转操作)
    head.next = null; //防止链表循环，需要将head.next设置为空
    return newHead;// 每层递归函数都返回newHead，也就是最后一个节点
};

两数相加 #

  • 思路：

    1. 小学数学题，模拟相加操作。
    2. 需要遍历链表

/**
 * @param {ListNode} l1

 * @return {ListNode}
 */
var addTwoNumbers = function(l1, l2) {
    const l3 = new ListNode(0) // 定义一个链表来存放结果
    let p1 = l1 // 指向链表1的头部
    let p2 = l2 // 指向链表2的头部
    let p3 = l3 // 指向新建的链表
    let carry = 0 // 进位的数，即留到下一轮相加的数
    while(p1 || p2) {
        // 这两个三元判断是为了防止相加的两个数长度不同
        const v1 = p1 ? p1.val : 0
        const v2 = p2 ? p2.val : 0
        const val = v1 + v2 + carry
        carry = Math.floor(val / 10) // 相加后的十位数
        p3.next = new ListNode(val % 10) // 相加后的个位数
        // 指针继续往后走
        if(p1) p1 = p1.next
        if(p2) p2 = p2.next
        p3 = p3.next
    }
    // 处理特殊情况：如[2,2,7] + [5,6,4]这种加到最后一个还有进位的情况
    if(carry) {
        p3.next = new ListNode(carry)
    }
    return l3.next // 第一个节点为空节点
};

  • 思路

    1. 先找出所有的不包含重复字符的字串。
    2. 找出长度最大的那个子串，返回其长度即可。

  • 解题步骤

    1. 用双指针维护一个滑动窗口，来剪切子串。
    2. 不断移动右指针，遇到重复字符，就把左指针移动到重复字符的下一位。
    3. 过程中，记录所有窗口的长度，并返回最大值。

    /**
 * @param {string} s
 * @return {number}
 */

    let l = 0 // 左指针
    let res = 0 // 最大长度
    const map = new Map()
    for(let r = 0; r < s.length; r++) {
        if(map.has(s[r]) && map.get(s[r]) >= l) { // 保证字符在滑动窗口里 "abbcdea"
            l = map.get(s[r]) + 1 // 遇到重复字符串，就把左指针移动到重复字符的下一位

        map.set(s[r], r) // 把当前字符及下标位置放进字典里
    }
    return res
};

最小覆盖子串（76） #

  • 思路

    1. 先和造出所有的包含T的子串。
    2. 找出长度最小的那个子串，返回即可。

  • 解题步骤

    1. 用双指针维护一个滑动窗口。
    2. 移动右指针，找到包含T的子串，再移动左指针，尽量减少包含T的子串的长度。
    3. 循环上述过程，找出包含T的最小子串。

  • 代码

    • 时间：O(m+n)
    • 空间：O(m)

    /**
 * @param {string} s
 * @param {string} t
 * @return {string}
 */
var minWindow = function(s, t) {
    let left = 0
    let right = 0
    const need = new Map() // 滑动窗口所需要的字符及个数
    for(let c of t) {
        need.set(c, need.has(c) ? need.get(c) + 1 : 1)
    }
    let needType = need.size // 需要的类型数量
    let res = '' // 最小子串
    while(right < s.length) {
        const current = s[right] // 右指针当前字符
        if(need.has(current)) {
            need.set(current, need.get(current) -1)
            if(need.get(current) === 0) needType -= 1
        }
        // 当右指针移动到满足含有所有字串后，移动左指针尽可能的缩小字符串长度
        while(needType === 0) {
            const newRes = s.substring(left, right + 1)
            if(!res || newRes.length < res.length) res = newRes // 更新最小子串
            const currentLeft = s[left] // 左指针当前字符
            if(need.has(currentLeft)) {
                need.set(currentLeft, need.get(currentLeft) + 1) // 如果左指针当前值存在于所需子字符串中，则对应的需求列表中该字符就得 +  1
                if(need.get(currentLeft) === 1) needType += 1 // 对应的所需字符种类也要增加
            }
            left += 1 // 移动左指针
        }
        right += 1 // 移动右指针
    }
    return res
};

二叉树的最大深度（104） #

  • 思路：

    1. 求最大深度，考虑使用深度优先遍历。
    2. 在深度优先遍历过程中，记录每个节点所在的层级，找出最大的层级即可。

  • 解题步骤：

    1. 新建一个变量，记录最大深度。
    2. 深度优先遍历整棵树，并记录每个节点的层级，同时不断刷新最大深度这个变量。
    3. 遍历结束后，返回最大深度这个变量。

  • 代码

    • 空间：O(n)~O(logN)
    • 时间：O(n)
    js
    /**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var maxDepth = function(root) {
    // 使用深度优先遍历
    // 在深度优先的遍历过程中，记录每个节点所在的层级，找出最大即可
    let res = 0 // 层级
    const dfs = (n, level) => {
        if(!n) return
        if(!n.left && !n.right) { // 仅当是叶子节点才计算刷新
            res = Math.max(res, level)
        }
        dfs(n.left, level +1) // 访问子节点 层级加一
        dfs(n.right, level + 1) // 访问子节点 层级加一
    }
    dfs(root, 1)
    return res
};

  • 解题步骤

    2. 遍历字符串，并沿着图走，如果到了某个节点无路可走就返回 false。
    3. 遍历结束，如果走到 3/5/6，就返回 true，否则返回 false。

    const graph = {//点和边构成的临接表
        0:{ 'blank': 0, 'sign': 1, '.': 2, 'digit': 6 }, // sign 操作符
        1:{ 'digit': 6, '.': 2 },
        2:{ 'digit': 3 },
        3:{ 'digit': 3, 'e': 4 },

    }

        if (c >= '0' && c <= '9') {
            c = 'digit';
        } else if (c === ' ') {
            c = 'blank';
        } else if(c === '+' ||  c === '-') {
            c = 'sign'
        } else if(c === 'E' || c === 'e') {

太平洋大西洋水流问题（417）

  • 思路

    1. 把矩阵想象成图。
    2. 从海岸线逆流而上遍历图，所到之处就是可以流到某个大洋的坐标。

  • 解题步骤

    1. 新建两个矩阵，分别记录能流到两个大洋的坐标。
    2. 从海岸线，多管齐下，同时深度优先遍历图，过程中填充上述矩阵。
    3. 遍历两个矩阵，找到能流到两个大洋的坐标。

  • 代码

    js
    /**
 * @param {number[][]} heights
 * @return {number[][]}
 */
var pacificAtlantic = function(heights) {
    if(!heights || !heights[0]) { return [] }
    const m = heights.length // 行数
    const n = heights[0].length // 列数
    // 新建两个矩阵, 分别代表太平洋 和 大西洋
    const flow1 = Array.from({ length: m }, () => new Array(n).fill(false))
    const flow2 = Array.from({ length: m }, () => new Array(n).fill(false))

    // 深度优先遍历
    const dfs = (r,c, flow) => {
        flow[r][c] = true
        // 上下左右四个节点
        const arr = [[r-1, c], [r + 1, c], [r, c - 1], [r, c + 1]]
        arr.forEach(([nr, nc]) => {
            if(
                // 保证节点都在矩阵中
                nr >= 0 && nr < m &&
                nc >= 0 && nc < n &&
                // 保证节点都是没有访问过的, 防止死循环(未访问节点都为false)
                !flow[nr][nc] && 
                // 保证是 ‘逆流而上’
                heights[nr][nc] >= heights[r][c]
            ) {
                dfs(nr, nc, flow)
            }
        })
    }

    // 沿着海岸线 逆流而上
    for(let r = 0; r < m; r++) {
        dfs(r, 0, flow1) // 太平洋 第一列 ----> 最左边
        dfs(r, n - 1, flow2) // 大西洋 最后一列 ----> 最右边
    }
    for(let c = 0; c < n; c++) {
        dfs(0, c, flow1) // 太平洋 第一行 -----> 最上边
        dfs(m - 1, c, flow2) // 大西洋 最后一行 ----> 最下面
    }

    // 收集能流到两个大洋里的坐标
    const res = []
    // 循环给定的矩阵，收集结果
    for(let r = 0; r<m; r++) {
        for(let c = 0; c < n; c++) {
            if(flow1[r][c] && flow2[r][c]) {
                res.push([r,c])
            }
        }
    }
    return res
};

堆 #

堆是什么？ #

  • 堆是一种特殊的完全二叉树
  • 所有的节点都大于等于（最大堆）或小于等于（最小堆）它的子节点。

JS 中的堆

  • JS 中通常用数组表示堆
  • 左侧子节点的位置是 2 * index + 1
  • 右侧子节点的位置是 2* index + 2
  • 父节点的位置是 (index - 1) / 2

堆的应用 #

  • 堆能高效、快速地找出最小值和最小值，时间复杂度：O(1)。
  • 找出第 K 个最大（小）元素
    1. 构建一个最小堆，并将元素依次插入堆中
    2. 当堆的容量超过K，就删除堆顶
    3. 插入结束后，堆顶就是第 K 个最大元素

    • 在类里，声明一个数组，用来装元素

    • 主要方法：插入、删除堆顶、获取堆顶、获取堆大小

    • 插入：

      1. 将值插入堆的底部，即数组的尾部。

      2. 然后上移：将这个值和它的父节点进行交换，直到父节点小于等于这个插入的值。

      3. 大小为 k 的堆中插入元素的时间复杂度为O(logk)。

选择排序 #

  • 思路：

    1. 找到数组中的最小值，选中它并将其放置在第一位。
    2. 接着找到第二小的值，选中它并将其放置在第二位。
    3. 以此类推，执行 n -1 轮

    Array.prototype.selectionSort = function () {
    for(let i = 0; i < this.length - 1; i++) {
        let indexMin = i // 把第一个没有排序过的元素设置为最小值
        // 遍历每个没有排序过的元素
        for(let j = i; j < this.length; j++) {
            if(this[j] < this[indexMin]) {

        if(indexMin !== i) {
            const temp = this[i]
            this[i] = this[indexMin]
            this[indexMin] = temp
        }
    }
}

    • 时间复杂度为：O(n^2)

    Array.prototype.mergeSort = function() {
    const merge = arr => {
        const res = []
        if(arr.length === 1) { return arr }

}

  • 时间复杂度：

    Array.prototype.binarySearch = function(item) {
    let low = 0 // 最小下标
    let hight = this.length - 1 // 最大下标
    while(low <= hight) {
        const mid = Math.floor((low + hight) / 2)
        const element = this[mid]
        if(element < item) {
            low = mid + 1
        } else if(element > item) {
            hight = mid - 1
        } else {
            return mid
        }
    }
    return -1
}

 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} p
 * @param {TreeNode} q
 * @return {boolean}
 */
var isSameTree = function(p, q) {
    if(!p && !q) {
        return true
    }
    if(p && q && p.val === q.val && isSameTree(p.left, q.left) && isSameTree(p.right, q.right)) {
        return true
    }
    return false
};

对称二叉树（101） #

  • 思路

    1. 转化为：左右子树是否镜像。
    2. 分解为：树1的左子树合数2的右子树是否镜像，树1的右子树合树2的左子树是否镜像。
    3. 符合“分、解、合”特性，选择分而治之。

  • 解题步骤

    1. 分：获取两个树的左子树和右子树。
    2. 解：递归地判断树1的左子树和树2的右子树是否镜像，树1的右子树和树2的左子树是否镜像。
    3. 合：如果上述都成立，且根节点值也相同，两个树就镜像。

  • 代码

    js
    /**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {boolean}
 */
var isSymmetric = function(root) {
     if(!root) { return true }
     const isMirror = (l, r) => {
         if(!l && !r) { return true }
         if(l && r && l.val === r.val && isMirror(l.left, r.right) && isMirror(l.right, r.left)) {
             return true
         }
         return false
     }
     return isMirror(root.left, root.right)
};

动态规划 #

动态规划是什么？ #

  • 动态规划是算法设计中的一种方法
  • 它将一个问题分解为相互重叠的子问题，通过反复求解子问题，来解决原来的问题。

斐波那契数列的应用 #

  • n = 0，F(n) = 0
  • n =1，F(n) = 1
  • 定义子问题：F(n) = F(n - 1) + F(n - 2) （相互重叠，反复求解子问题）
  • 反复执行：从 2 循环到 n，执行上述公式。

动态规划VS分而治之 #

  • 区别：在于子问题是否重叠，和子问题是否独立

爬楼梯（70） #

  • 思路：

    1. 爬到第 n 阶可以在第 n-1 阶爬 1 个台阶，或者在第 n-2 阶爬 2 个台阶。
    2. 则 F(n) = F(n-1) + F(n-2)
    3. 使用动态规划

  • 解题步骤：

    1. 定义子问题：F(n) = F(n-1) + F(n-2)
    2. 反复执行：从 2 循环到 n, 执行上述公式

  • 代码：

    js
    /**
 * @param {number} n
 * @return {number}
 */
var climbStairs = function(n) {
    if(n < 2) { return 1 }
    const dp = [1,1]
    for(let i = 2; i <= n; i++) {
        dp[i] = dp[i-1] + dp[i-2]
    }
    return dp[n]
};

  • 优化（空间）版代码

    js
    /**
 * @param {number} n
 * @return {number}

    let a = 1
    let b = 1
    for(let i = 2; i <= n; i++) {
        const t = a
        a = b
        b = b + t
    }
    return b
};

  • 解题步骤：

    1. 定义子问题：f(k) = max(f(k - 2) + Ak, f(k - 1))
    2. 反复执行：从2循环到 n, 执行上述公式

  • 代码

    js
    /**
 * @param {number[]} nums
 * @return {number}
 */
var rob = function(nums) {
    if(nums.length === 0) { return 0 }
    let dp0 = 0
    let dp1 = nums[0]
    for(let i = 2; i <= nums.length; i++) {

        dp1 = dp2
    }

回溯算法 #

回溯算法是什么？ #

  • 回溯算法是算法设计中的一种方法。
  • 回溯算法是一种渐进式寻找并构建问题解决方式的策略。（不是一步到位）
  • 回溯算法会先从一个可能的动作开始解决问题，如果不行，就回溯并选择另外一个动作，直到将问题解决。

什么问题适合用回溯算法解决？ #

  • 有很多路

  • 这些路里，有死路，也有出路

  • 通常需要递归来模拟所有的路

  • 例子：

    • 全排列：
      1. 用递归模拟出全部的情况。
      2. 遇到包含重复元素的情况，就回溯。
      3. 收集所有到达递归终点的情况，并返回。

全排列（46） #

  • 思路：

    1. 要求：（1）所有排列情况。（2）没有重复元素。
    2. 有出路，有死路
    3. 考虑使用回溯算法

  • 解题步骤：

    1. 用递归模拟所有情况。
    2. 遇到包含重复元素的情况，就回溯
    3. 收集所有到达终点的情况，并返回。

  • 代码：

    1. 时间：O（n!）循环每次都在减少
    2. 空间：O（n）递归的层数
    js
    /**
 * @param {number[]} nums
 * @return {number[][]}
 */
var permute = function(nums) {
    const res = []
    const backtrack = (path) => {
        if(path.length === nums.length) {
            res.push(path)
            return
        }
        nums.forEach(n => {
            if(path.includes(n)) { return } // 死路
            backtrack(path.concat(n))
        })
    }
    backtrack([])
    return res
};

子集（78） #

  • 思路：

    1. 要求：（1）所有子集。（2）没有重复元素
    2. 有出路，有死路。
    3. 考虑使用回溯算法。

  • 解题步骤：

    1. 用递归模拟出所有的情况。
    2. 保证接的数字都是后面的数字。
    3. 收集所有到达终点的情况，并返回。

  • 代码

    1. 时间：O（2^n）因为每个元素都有两种可能（存在或不存在）
    2. 空间：O（n）
    js
    /**
 * @param {number[]} nums
 * @return {number[][]}
 */
var subsets = function(nums) {
    const res = []
    const backstrack = (path, l, start) => {
        if(path.length === l) { //到达终点
            res.push(path)
            return
        }
        for(let i = start; i < nums.length; i++) {
            backstrack(path.concat(nums[i]), l, i+1) // i+1 保证后面的数字都是当前数字后面的数字
        }
    }
    for(let i = 0; i <= nums.length; i++) {// 所有子集情况， 回溯调用
        backstrack([], i, 0) // 输入路径和长度
    }
    return res
};
